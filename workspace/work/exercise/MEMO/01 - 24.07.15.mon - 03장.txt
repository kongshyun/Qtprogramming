	복합대입연산자
		+=
		
	<비트연산자>
	EX)
	#include <stdio.h>

	int main(void)
	{
		int a = 10;			// 비트열 00000000 00000000 00000000 00001010
		int b = 12;			// 비트열 00000000 00000000 00000000 00001100

		printf("a & b : %d\n", a & b);
		printf("a ^ b : %d\n", a ^ b);
		printf("a | b : %d\n", a | b);
		printf("~a : %d\n", ~a);
		printf("a << 1 : %d\n", a << 1);
		printf("a >> 2 : %d\n", a >> 2);

		return 0;
	}
	
	--> 결과
	a & b : 8
	a ^ b : 6
	a | b : 14
	~a : -11
	a << 1 : 20
	a >> 2 : 2

	<연산자 우선순위>
		포인터와 관련
		1차연산자, 단항연산자 중요
			- 1차연산자 : () , [], ' , ->
			- 단항연산자: -, ++, --, ~, !, *(포인터연산자), &(어드레스 연산자), sizeof(type)
				단한연산자는 방향이 <---- 이방향으로 계산!
